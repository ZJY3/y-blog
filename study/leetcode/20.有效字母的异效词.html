    有效的字母异效词

    zjy 2023/4/11 字符串ASCII编码
    package cn.zjy.leetcode.demo;

/**
 * @author ZhangJiaYi
 * @time 2023/04/11 09:52:49
 * @description demo020 有效的字母异效词
 */
public class Demo020 {
    public static void main(String[] args) {
        System.out.println(isAnagram("abac", "aad"));

    }

    /**
     * 先将s的字符计数，然后在遍历t串 每次出现将数组次数-1 只有最差情况才会将t串全部遍历
     * Time: O(n) 遍历字符串计数
     * Space: O(n) 数组存储字符的ASCII编码出现次数
     *
     * @param s 字符串
     * @param t 字符串
     * @return boolean
     */
    public static boolean isAnagram(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] num = new int[26];
        for (char c : s.toCharArray()) {
            num[c - 'a']++;
        }
        for (char c : t.toCharArray()) {
            if (--num[c - 'a'] < 0) {
                return false;
            }
        }
        return true;
    }

    /**
     * 先将两个字符串存入 再去判断数量 相当于直接是方案1的最差情况 而且还调用了charAt浪费了时间和空间
     * Time: O(n)
     * Space: O(n)
     *
     * @param s s
     * @param t t
     * @return boolean
     */
    public static boolean isAnagram2(String s, String t) {
        if (s.length() != t.length()) {
            return false;
        }
        int[] nums = new int[26];
        for (int i = 0; i < s.length(); i++) {
            nums[s.charAt(i) - 'a']++;
            nums[t.charAt(i) - 'a']--;
        }
        for (int num : nums) {
            if (num < 0) {
                return false;
            }
        }
        return true;

    }
}
